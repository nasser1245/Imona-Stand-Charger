CREATE TABLE "Achievement" (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`StartDT`	TEXT NOT NULL,
	`EndDT`	TEXT NOT NULL,
	`VideoID`	INTEGER NOT NULL,
	`StartFrame`	INTEGER NOT NULL,
	`EndFrame`	INTEGER NOT NULL,
	`AchieveType`	INTEGER NOT NULL,
	`Fingerprint`	TEXT,
	`Deleted`	INTEGER NOT NULL DEFAULT 0
);
CREATE TABLE "Picture" (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`Address`	INTEGER NOT NULL,
	`AchievementID`	INTEGER NOT NULL,
	`CaptureDT`	INTEGER NOT NULL,
	`Deleted`	INTEGER NOT NULL DEFAULT 0
);
CREATE TABLE "ShelfSocket" (
	`ShelfID`	INTEGER NOT NULL,
	`SocketID`	INTEGER NOT NULL,
	`Active`	INTEGER NOT NULL DEFAULT 1
);
CREATE TABLE "ShelfStatus" (
	`ID`	INTEGER NOT NULL,
	`SocketType`	INTEGER DEFAULT 0,
	`AchievementID`	INTEGER NOT NULL DEFAULT 0,
	PRIMARY KEY(`ID`)
);
CREATE TABLE "Socket" (
	`ID`	INTEGER NOT NULL UNIQUE,
	`name`	TEXT NOT NULL UNIQUE,
	PRIMARY KEY(`ID`)
);
CREATE TABLE "Video" (
	`ID`	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE,
	`Address`	TEXT NOT NULL,
	`StartCaptureDT`	TEXT NOT NULL,
	`EndCaptureDT`	TEXT NOT NULL,
	`Deleted`	INTEGER NOT NULL DEFAULT 0
);
